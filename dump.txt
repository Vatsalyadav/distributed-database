CREATE TABLE students (  BannerID varchar(30) NOT NULL,  name varchar(10) NOT NULL,  weight INT  eyecolor varchar(30) , PRIMARY KEY (BannerID) );
CREATE TABLE courses (  BannerID varchar(30) NOT NULL,  BannerID varchar(30) NOT NULL,  courseID varchar(10) NOT NULL,  courseName varchar(30) , PRIMARY KEY (BannerID), FOREIGN KEY (BannerID) REFERENCES students(BannerID) );
